# Build Command para configurar no Render
# Copie e cole este comando no campo "Build Command" do Render

# ⚠️ IMPORTANTE: Com PostgreSQL, não precisamos criar arquivo de banco durante o build
# O PostgreSQL será criado automaticamente quando o Django iniciar no runtime
# Use --skip-checks para evitar que o Django tente acessar o banco durante o build

pip install -r requirements.txt && python manage.py migrate --noinput --skip-checks && python manage.py collectstatic --noinput --clear --skip-checks
